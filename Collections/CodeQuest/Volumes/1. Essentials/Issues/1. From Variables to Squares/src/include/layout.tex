\newcommand{\BlocklyGhost}[4]{%
  \begin{scope}[shift={({#1},{#2})}, scale=#3]
    \fill[#4, opacity=0.11, rounded corners=3pt] (0,0) rectangle (3.6,1.0);
    \fill[#4, opacity=0.11, rounded corners=2pt] (0.8,1.0) rectangle (1.8,1.32);
    \fill[cqbg] (1.15,0) rectangle (1.95,0.22);
    \draw[#4, opacity=0.30, line width=0.5pt, rounded corners=3pt] (0,0) rectangle (3.6,1.0);
    \draw[#4, opacity=0.30, line width=0.5pt, rounded corners=2pt] (0.8,1.0) rectangle (1.8,1.32);
  \end{scope}
}
