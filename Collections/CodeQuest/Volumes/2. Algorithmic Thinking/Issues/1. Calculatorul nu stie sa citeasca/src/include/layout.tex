\newcommand{\DigitScannerGhost}[4]{%
  \begin{scope}[shift={({#1},{#2})}, scale=#3]
    \draw[#4, opacity=0.22, line width=0.5pt, rounded corners=3pt] (0,0) rectangle (4.6,1.35);
    \draw[#4, opacity=0.14, line width=0.5pt] (0.35,0.24) -- (4.25,0.24);
    \draw[#4, opacity=0.14, line width=0.5pt] (0.35,1.11) -- (4.25,1.11);
    \foreach \x/\d in {0.55/1,1.15/2,1.75/3,2.35/5,2.95/3,3.55/2,4.15/1} {
      \node[text=#4, opacity=0.18, font=\ttfamily\bfseries] at (\x,0.68) {\d};
    }
    \draw[#4, opacity=0.26, line width=1pt] (2.3,-0.2) -- (2.3,1.55);
    \draw[#4, opacity=0.18, line width=0.5pt] (1.1,-0.05) -- (3.5,1.4);
    \draw[#4, opacity=0.18, line width=0.5pt] (3.5,-0.05) -- (1.1,1.4);
  \end{scope}
}
